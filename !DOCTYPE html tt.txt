                <label for="imageInput">Adicionar imagem (opcional)</label>
                <input type="file" id="imageInput" accept="image/*">
                <img id="imagePreview" class="image-preview" style="display: none;">
            </div>
            <button class="btn btn-primary" onclick="submitPost()">Publicar 🚀</button>

                <input type="text" id="newCommentInput" placeholder="Adicione um comentário...">

    <script>
        // Configuração da API do Google Sheets
        const GOOGLE_APPS_SCRIPT_URL = 'INSIRA_SUA_URL_DO_GOOGLE_APPS_SCRIPT_AQUI';
        let currentPostId = null;
        let postsData = [];

        // Inicializar
        document.addEventListener('DOMContentLoaded', function() {
            const savedUserName = localStorage.getItem('userName');
            if (savedUserName) {
                document.getElementById('authorInput').value = savedUserName;
                document.getElementById('userName').textContent = savedUserName;
            }
            loadPosts();
        });

        // Manipular preview de imagem
        document.getElementById('imageInput').addEventListener('change', function(e) {
            const file = e.target.files[0];
            if (file) {
                const reader = new FileReader();
                reader.onload = function(event) {
                    const preview = document.getElementById('imagePreview');
                    preview.src = event.target.result;
                    preview.style.display = 'block';
                };
                reader.readAsDataURL(file);
            }
        });

        // Publicar post
        function submitPost() {
            const author = document.getElementById('authorInput').value.trim();
            const content = document.getElementById('contentInput').value.trim();
            const imageInput = document.getElementById('imageInput');

            if (!author) {
                showMessage('Por favor, digite seu nome', 'error');
                return;
            }
            if (!content) {
                showMessage('Por favor, compartilhe sua ideia', 'error');
                return;
            }

            localStorage.setItem('userName', author);
            document.getElementById('userName').textContent = author;

            const postData = {
                author: author,
                content: content,
                timestamp: new Date().toISOString(),
                likes: 0,
                reposts: 0,
                comments: []
            };

            // Se houver imagem, converter para base64
            if (imageInput.files.length > 0) {
                const reader = new FileReader();
                reader.onload = function(event) {
                    postData.image = event.target.result;
                    savePost(postData);
                };
                reader.readAsDataURL(imageInput.files[0]);
            } else {
                savePost(postData);
            }
        }

        // Salvar post no Google Sheets
        function savePost(postData) {
            const messageDiv = document.getElementById('message');
            messageDiv.textContent = '📤 Publicando...';
            messageDiv.style.display = 'block';

            fetch(GOOGLE_APPS_SCRIPT_URL, {
                method: 'POST',
                body: JSON.stringify({
                    action: 'addPost',
                    post: postData
                })
            })
            .then(response => response.json())
            .then(data => {
                if (data.success) {
                    showMessage('✅ Ideia publicada com sucesso!', 'success');
                    document.getElementById('contentInput').value = '';
                    document.getElementById('imageInput').value = '';
                    document.getElementById('imagePreview').style.display = 'none';
                    loadPosts();
                } else {
                    showMessage('❌ Erro ao publicar: ' + data.message, 'error');
                }
            })
            .catch(error => {
                console.error('Erro:', error);
                showMessage('❌ Erro de conexão. Verifique a URL do Apps Script.', 'error');
            });
        }

        // Carregar posts
        function loadPosts() {
            const container = document.getElementById('postsContainer');
            container.innerHTML = '<div class="loading"><div class="spinner"></div><p>Carregando...</p></div>';

            fetch(GOOGLE_APPS_SCRIPT_URL, {
                method: 'POST',
                body: JSON.stringify({ action: 'getPosts' })
            })
            .then(response => response.json())
            .then(data => {
                if (data.success && data.posts) {
                    postsData = data.posts.reverse(); // Posts mais recentes primeiro
                    renderPosts();
                } else {
                    container.innerHTML = '<div class="empty-state"><p>Nenhuma ideia publicada ainda. Seja o primeiro! 🚀</p></div>';
                }
            })
            .catch(error => {
                console.error('Erro:', error);
                container.innerHTML = '<div class="empty-state"><p>Erro ao carregar posts. Verifique a URL do Apps Script.</p></div>';
            });
        }

        // Renderizar posts
        function renderPosts() {
            const container = document.getElementById('postsContainer');
            if (postsData.length === 0) {
                container.innerHTML = '<div class="empty-state"><p>Nenhuma ideia publicada ainda. Seja o primeiro! 🚀</p></div>';
                return;
            }

            container.innerHTML = postsData.map((post, index) => `
                <div class="post">
                    ${post.repostedBy ? `<div class="repost-badge">🔄 Repostado por ${post.repostedBy}</div>` : ''}
                    <div class="post-header">
                        <div>
                            <div class="post-author">${escapeHtml(post.author)}</div>
                            <div class="post-time">${formatTime(post.timestamp)}</div>
                        </div>
                        <button class="btn-small" onclick="repostModal(${index})">⤴️ Repostar</button>
                    </div>
                    <div class="post-content">${escapeHtml(post.content)}</div>
                    ${post.image ? `<img src="${post.image}" class="post-image" alt="Imagem do post">` : ''}
                    <div class="post-actions">
                        <button class="btn-small ${isLiked(post, index) ? 'active' : ''}" onclick="toggleLike(${index})">
                            ❤️ <span id="likes-${index}">${post.likes || 0}</span>
                        </button>
                        <button class="btn-small" onclick="openCommentsModal(${index})">
                            💬 <span id="comments-${index}">${(post.comments || []).length}</span>
                        </button>
                        <button class="btn-small" onclick="copyLink('${encodeURIComponent(JSON.stringify(post))}')">
                            🔗 Copiar
                        </button>
                    </div>
                    ${(post.comments && post.comments.length > 0) ? `
                        <div class="comments-section">
                            <strong style="color: #667eea;">Comentários recentes:</strong>
                            ${post.comments.slice(0, 2).map(comment => `
                                <div class="comment">
                                    <div class="comment-author">${escapeHtml(comment.author)}</div>
                                    <div class="comment-text">${escapeHtml(comment.text)}</div>
                                    <div class="comment-time">${formatTime(comment.timestamp)}</div>
                                </div>
                            `).join('')}
                            ${post.comments.length > 2 ? `<p style="text-align: center; color: #999; font-size: 13px; margin-top: 10px;">+${post.comments.length - 2} mais comentários</p>` : ''}
                        </div>
                    ` : ''}
                </div>
            `).join('');
        }

        // Curtir post
        function toggleLike(index) {
            const post = postsData[index];
            post.likes = (post.likes || 0) + 1;
            const key = `liked-post-${index}`;
            if (!localStorage.getItem(key)) {
                localStorage.setItem(key, 'true');
                updatePost(index);
            }
        }

        // Verificar se usuário curtiu
        function isLiked(post, index) {
            return localStorage.getItem(`liked-post-${index}`) === 'true';
        }

        // Repostar modal
        function repostModal(index) {
            const post = postsData[index];
            const userName = document.getElementById('authorInput').value || 'Usuário';
            if (!userName) {
                showMessage('Por favor, digite seu nome primeiro', 'error');
                return;
            }
            const confirmRepost = confirm(`Deseja repostar a ideia de "${post.author}"?`);
            if (confirmRepost) {
                repost(index);
            }
        }

        // Repostar
        function repost(index) {
            const post = JSON.parse(JSON.stringify(postsData[index]));
            const userName = document.getElementById('authorInput').value;
            post.repostedBy = userName;
            post.repostedAt = new Date().toISOString();

            fetch(GOOGLE_APPS_SCRIPT_URL, {
                method: 'POST',
                body: JSON.stringify({
                    action: 'addPost',
                    post: post
                })
            })
            .then(response => response.json())
            .then(data => {
                if (data.success) {
                    showMessage('✅ Ideia repostada com sucesso!', 'success');
                    loadPosts();
                } else {
                    showMessage('❌ Erro ao repostar', 'error');
                }
            })
            .catch(error => {
                console.error('Erro:', error);
                showMessage('❌ Erro de conexão', 'error');
            });
        }

        // Abrir modal de comentários
        function openCommentsModal(index) {
            currentPostId = index;
            const post = postsData[index];
            const container = document.getElementById('commentsContainer');

            if (post.comments && post.comments.length > 0) {
                container.innerHTML = post.comments.map(comment => `
                    <div class="comment">
                        <div class="comment-author">${escapeHtml(comment.author)}</div>
                        <div class="comment-text">${escapeHtml(comment.text)}</div>
                        <div class="comment-time">${formatTime(comment.timestamp)}</div>
                    </div>
                `).join('');
            } else {
                container.innerHTML = '<p style="color: #999; text-align: center;">Nenhum comentário ainda. Seja o primeiro!</p>';
            }

            document.getElementById('commentsModal').classList.add('active');
            document.getElementById('newCommentInput').focus();
        }

        // Fechar modal
        function closeCommentsModal() {
            document.getElementById('commentsModal').classList.remove('active');
            document.getElementById('newCommentInput').value = '';
            currentPostId = null;
        }

        // Adicionar comentário
        function addComment() {
            const commentText = document.getElementById('newCommentInput').value.trim();
            const author = document.getElementById('authorInput').value.trim();

            if (!author) {
                showMessage('Por favor, digite seu nome', 'error');
                return;
            }
            if (!commentText) {
                showMessage('Por favor, escreva um comentário', 'error');
                return;
            }

            const post = postsData[currentPostId];
            if (!post.comments) post.comments = [];

            post.comments.push({
                author: author,
                text: commentText,
                timestamp: new Date().toISOString()
            });

            updatePost(currentPostId);
            document.getElementById('newCommentInput').value = '';
            openCommentsModal(currentPostId);
        }

        // Atualizar post
        function updatePost(index) {
            fetch(GOOGLE_APPS_SCRIPT_URL, {
                method: 'POST',
                body: JSON.stringify({
                    action: 'updatePost',
                    index: index,
                    post: postsData[index]
                })
            })
            .then(response => response.json())
            .then(data => {
                if (data.success) {
                    renderPosts();
                } else {
                    console.error('Erro ao atualizar:', data.message);
                }
            })
            .catch(error => console.error('Erro:', error));
        }

        // Copiar link
        function copyLink(postData) {
            const link = `${window.location.href}?post=${postData}`;
            navigator.clipboard.writeText(link).then(() => {
                showMessage('✅ Link copiado!', 'success');
            });
        }

        // Mostrar mensagem
        function showMessage(text, type) {
            const messageDiv = document.getElementById('message');
            messageDiv.textContent = text;
            messageDiv.className = type;
            messageDiv.style.display = 'block';
            setTimeout(() => {
                messageDiv.style.display = 'none';
            }, 3000);
        }

        // Formatar tempo
        function formatTime(timestamp) {
            const date = new Date(timestamp);
            const now = new Date();
            const diffMs = now - date;
            const diffMins = Math.floor(diffMs / 60000);
            const diffHours = Math.floor(diffMs / 3600000);
            const diffDays = Math.floor(diffMs / 86400000);

            if (diffMins < 1) return 'agora';
            if (diffMins < 60) return `${diffMins}m atrás`;
            if (diffHours < 24) return `${diffHours}h atrás`;
            if (diffDays < 7) return `${diffDays}d atrás`;
            return date.toLocaleDateString('pt-BR');
        }

        // Escapar HTML
        function escapeHtml(text) {
            const map = {
                '&': '&amp;',
                '<': '&lt;',
                '>': '&gt;',
                '"': '&quot;',
                "'": '&#039;'
            };
            return text.replace(/[&<>"']/g, m => map[m]);
        }

        // Fechar modal ao clicar fora
        document.getElementById('commentsModal').addEventListener('click', function(e) {
            if (e.target === this) {
                closeCommentsModal();
            }
        });

        // Atualizar posts a cada 5 segundos
        setInterval(loadPosts, 5000);
    </script>
